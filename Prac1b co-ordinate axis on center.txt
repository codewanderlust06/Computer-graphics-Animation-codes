#include <graphics.h>
#include <stdio.h>
#include <conio.h>

int main(void)
{
	/* request auto detection */
	int gdriver = DETECT, gmode;
	int xmax,ymax;

	/* initialize graphics mode */
	initgraph(&gdriver,&gmode, "C:\\TURBOC3\\BGI");
	xmax=getmaxx();
	ymax=getmaxy();
	line(xmax/2,0,xmax/2,ymax);
	line(0,ymax/2,xmax,ymax/2);
	getch();
	closegraph();
	return 0;
}
